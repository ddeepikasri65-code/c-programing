


#include<stdio.h>
#include<string.h>
struct department {
    int id;
    char Name[10];
    
};

struct Employee{
    int Id;
    char name[10];
    struct department details;
};

void printdepartmentDetails(struct Employee *Details){
    printf("Department ID: %d\n",Details->details.id);
    printf("Department Name: %s\n",Details->details.Name);
    printf("Employee ID: %d\n",Details->Id);
    printf("Employee Name: %s:",Details->name);
}

int main(){
    struct Employee Details;
    Details.details.id = 2656;
    strcpy(Details.details.Name,"ECE");
    Details.Id = 2029;
    strcpy(Details.name,"Deepika");
    printf("---Department Details---\n");
    printdepartmentDetails(&Details);
    return 0;
    
    
}

Output:

Department Name: ECE
Employee ID: 2029
