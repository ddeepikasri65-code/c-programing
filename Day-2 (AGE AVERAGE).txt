#include <stdio.h>

int main() {
        int n;
        printf("----AGE DETAILS----\n");
        printf("Enter the number of persons:\n");
        scanf("%d",&n);
        int arr[50];
        int sum = 0;
        float avg;
        for(int i=0;i<n;i++){
        printf("Enter the age:\n",i+1);
        scanf("%d",&arr[i]);
        sum = sum+arr[i];
        printf("Sum of age: %d",sum);
        }
        avg=sum/n;
        printf("The Average age is %.2f",avg);
        
    return 0;
}
